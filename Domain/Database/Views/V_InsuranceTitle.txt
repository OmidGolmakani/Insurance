SELECT        dbo.InsuranceTitle.Id, dbo.InsuranceTitle.Code, dbo.InsuranceTitle.[level], dbo.InsuranceTitle.Name, dbo.InsuranceTitle.Description, dbo.InsuranceTitle.Active, dbo.InsuranceTitle.IsDeleted, dbo.InsuranceTitle.DeletedDate, 
                         dbo.InsuranceTitle.DeletedBy, dbo.InsuranceTitle.CreatedDate, dbo.InsuranceTitle.CreatedBy, dbo.InsuranceTitle.LastModified, dbo.InsuranceTitle.LastModifiedBy, dbo.InsuranceTitle.LanguageId, dbo.InsuranceTitle.ParentId, 
                         dbo.Language.LanguageName, dbo.Language.AcceptLanguage
FROM            dbo.InsuranceTitle INNER JOIN
                         dbo.Language ON dbo.InsuranceTitle.LanguageId = dbo.Language.Id
